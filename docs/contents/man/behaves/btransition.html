Examples:


js

let myApp = new Application
let myText = new Label(myApp, model.label);

myText.setTransition("color", 2000);
myText.addEvent("transition", (e) => {
  console.log(e.data);

  switch (e.data.status) {
    case ETransition.START:
      myText.style.color = "red";
      break;

    case ETransition.INPROGRESS:
      myText.style.color = "white";
      break;

    case ETransition.FINNISH:
      myText.style.color = "";
      break;
  }
});

myText.startTransition("color");

css

[element-type][element-name=container][transition-name="color"][transition-status="start"] {
  transition-property: none;
  background-color: blue;
}

[element-type][element-name=container][transition-name="color"][transition-status="inprogress"] {
  transition-property: background-color, color;
  transition-duration: 2s;
  background-color: red;
}
