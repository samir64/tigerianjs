class BHiFamily extends Behavior {
  constructor () {
    super();

    this.defineMethod("config", (that, family) => {
      that.defineMethod("sayHi", (name) => {
        console.log(`Hi, ${name} ${family}.`);
      })
    });
  }
}

class HiFamily extends Tigerian {
  constructor (family) {
    super ();

    this.config(BHiFamily, family);
  }
}

let mc1 = new HiFamily("Gonjishke");
mc1.sayHi("Jack");
mc1.sayHi("Joe");
